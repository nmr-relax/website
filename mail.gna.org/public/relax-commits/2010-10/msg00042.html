+    def get_spheroid(self):
+        """Return all the diffusion tensor info about the {Dpar, Dper, 
theta, phi} = {1e7, 2e7, 1.5, 2} spheroid tensor."""
+

+        Dpar = 1e7
+        Dper = 4e7

+        Da = -3e7
+        theta = 0.5
+        phi = 1.0

+                   [   880899.67391431,  -1924800.90303685,  
39849887.44900668]], float64)
+
