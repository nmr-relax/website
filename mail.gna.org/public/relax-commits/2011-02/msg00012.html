+def num_spins(data):
+    """Determine the number of spins in the given data.
+
+    @param data:    An element of data from bmrblib.
+    @type data:     dict
+    @return:        The number of spins.
+    @rtype:         int
+    """
+
+    # The number of spins.
+    N = 0
+    if data['data_ids']:
+        N = len(data['data_ids'])
+    elif data['entity_ids']:
+        N = len(data['entity_ids'])
+    elif data['res_nums']:
+        N = len(data['res_nums'])
+    elif data['s2']:
+        N = len(data['s2'])
+
+    # Return the number.
+    return N
