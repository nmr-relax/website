+    def test_copy_between_pipes(self):

+        The function used is generic_fns.molecule.copy().

mol_to='#New mol')
+

+        relax_data_store['orig'].mol[0].res[0].spin[0].num = 222
+        relax_data_store['orig'].mol[0].res[0].spin[0].x = 2
+
