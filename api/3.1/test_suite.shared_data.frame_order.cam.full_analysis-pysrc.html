 22  """Script for black-box Frame Order analysis. 
 23   
 24  This script should be run from the directory where it is found with the commands: 
 25   
 26  $ ../../../../../relax full_analysis.py 
 27   
 28  The free rotor pseudo-elliptic cone model is not used in this script as the cone X and Y opening angles cannot be differentiated with simply RDC and PCS data, hence this model is perfectly approximated by the free rotor isotropic cone. 
 29  """ 
 30   
 31  # Python module imports. 
 32  from numpy import array 
 33  from time import asctime, localtime 
 34   
 35  # relax module imports. 
 36  from auto_analyses.frame_order import Frame_order_analysis 
 37   
 38   
 39  # Analysis variables. 
 40  ##################### 
 41   
 42  # The grid search size (the number of increments per dimension). 
 43  GRID_INC = 11 
 44   
 45  # The more precise grid search size for the initial rigid model (the number of increments per dimension). 
 46  GRID_INC_RIGID = 31 
 47   
 48  # The number of Sobol' points for the PCS numerical integration in the grid searches. 
 49  NUM_INT_PTS_GRID = 50 
 50   
 51  # The list of the number of Sobol' points for the PCS numerical integration to use iteratively in the optimisations after the grid search (for the PCS data subset). 
 52  NUM_INT_PTS_SUBSET = [100] 
 53   
 54  # The minimisation function tolerance cutoff to terminate optimisation (for the PCS data subset, see the minimise user function). 
 55  FUNC_TOL_SUBSET = [1e-2] 
 56   
 57  # The list of the number of Sobol' points for the PCS numerical integration to use iteratively in the optimisations after the grid search (for all PCS and RDC data). 
 58  NUM_INT_PTS_FULL = [100, 1000, 10000] 
 59   
 60  # The minimisation function tolerance cutoff to terminate optimisation (for all PCS and RDC data, see the minimise user function). 
 61  FUNC_TOL_FULL = [1e-2, 1e-3, 1e-4] 
 62   
 63  # The optimisation technique. 
 64  MIN_ALGOR = 'simplex' 
 65   
 66  # The number of Monte Carlo simulations to be used for error analysis at the end of the protocol. 
 67  MC_NUM = 100 
 68   
 69  # The number of Sobol' points for the PCS numerical integration during Monte Carlo simulations. 
 70  MC_INT_PTS = 100 
 71   
 72  # The minimisation function tolerance cutoff to terminate optimisation during Monte Carlo simulations. 
 73  MC_FUNC_TOL = 1e-2 
 74   
 75  # The frame order models to use. 
 76  MODELS = [ 
 77      'rigid', 
 78      'free rotor', 
 79      'rotor', 
 80      'iso cone, torsionless', 
 81      'iso cone, free rotor', 
 82      'iso cone', 
 83      'pseudo-ellipse, torsionless', 
 84      'pseudo-ellipse' 
 85  ] 
 86   
 87  # Set up the base data pipes. 
 88  ############################# 
 89   
 90  # The data pipe bundle to group all data pipes. 
 91  PIPE_BUNDLE = "Frame Order (%s)" % asctime(localtime()) 
 92   
 93  # Create the base data pipe containing only a subset of the PCS data. 
 94  SUBSET = "Data subset - " + PIPE_BUNDLE 
 95  pipe.create(pipe_name=SUBSET, pipe_type='frame order', bundle=PIPE_BUNDLE) 
 96   
 97  # Read the structures. 
 98  structure.read_pdb('1J7O_1st_NH.pdb', dir='..', set_mol_name='N-dom') 
 99  structure.read_pdb('1J7P_1st_NH_rot.pdb', dir='..', set_mol_name='C-dom') 
100   
101  # Set up the 15N and 1H spins. 
102  structure.load_spins(spin_id='@N', ave_pos=False) 
103  structure.load_spins(spin_id='@H', ave_pos=False) 
104  spin.isotope(isotope='15N', spin_id='@N') 
105  spin.isotope(isotope='1H', spin_id='@H') 
106   
107  # Define the magnetic dipole-dipole relaxation interaction. 
108  interatom.define(spin_id1='@N', spin_id2='@H', direct_bond=True) 
109  interatom.set_dist(spin_id1='@N', spin_id2='@H', ave_dist=1.041 * 1e-10) 
110  interatom.unit_vectors() 
111   
112  # The lanthanides and data files. 
113  ln = ['dy', 'tb', 'tm', 'er'] 
114  pcs_files = [ 
115      'pcs_dy.txt', 
116      'pcs_tb.txt',  
117      'pcs_tm.txt',  
118      'pcs_er.txt' 
119  ] 
120  pcs_files_subset = [ 
121      'pcs_dy_subset.txt',  
122      'pcs_tb_subset.txt',  
123      'pcs_tm_subset.txt',  
124      'pcs_er_subset.txt'  
125  ] 
126  rdc_files = [ 
127      'rdc_dy.txt',  
128      'rdc_tb.txt',  
129      'rdc_tm.txt',  
130      'rdc_er.txt'  
131  ] 
132   
133  # The spectrometer frequencies. 
134  pcs_frq = [ 
135      700.0000000,    # Dy3+. 
136      700.0000000,    # Tb3+. 
137      700.0000000,    # Tm3+. 
138      700.0000000,    # Er3+. 
139  ] 
140  rdc_frq = [ 
141      900.00000000,   # Dy3+. 
142      900.00000000,   # Tb3+. 
143      900.00000000,   # Tm3+. 
144      900.00000000,   # Er3+. 
145  ] 
146   
147  # Loop over the alignments. 
148  for i in range(len(ln)): 
149      # Load the RDCs. 
150      rdc.read(align_id=ln[i], file=rdc_files[i], dir='.', spin_id1_col=1, spin_id2_col=2, data_col=3, error_col=4) 
151   
152      # The PCS (only a subset of ~5 spins for fast initial optimisations). 
153      pcs.read(align_id=ln[i], file=pcs_files_subset[i], dir='.', mol_name_col=1, res_num_col=2, spin_name_col=5, data_col=6, error_col=7) 
154   
155      # The temperature and field strength. 
156      spectrometer.temperature(id=ln[i], temp=303.0) 
157      spectrometer.frequency(id=ln[i], frq=rdc_frq[i], units="MHz") 
158   
159  # Load the N-domain tensors (the full tensors). 
160  script('../tensors.py') 
161   
162  # Define the domains. 
163  domain(id='N', spin_id=":1-78") 
164  domain(id='C', spin_id=":80-148") 
165   
166  # The tensor domains and reductions. 
167  full = ['Dy N-dom', 'Tb N-dom', 'Tm N-dom', 'Er N-dom'] 
168  red =  ['Dy C-dom', 'Tb C-dom', 'Tm C-dom', 'Er C-dom'] 
169  ids = ['dy', 'tb', 'tm', 'er'] 
170  for i in range(len(full)): 
171      # Initialise the reduced tensors (fitted during optimisation). 
172      align_tensor.init(tensor=red[i], align_id=ids[i], params=(0, 0, 0, 0, 0)) 
173   
174      # Set the domain info. 
175      align_tensor.set_domain(tensor=full[i], domain='N') 
176      align_tensor.set_domain(tensor=red[i], domain='C') 
177   
178      # Specify which tensor is reduced. 
179      align_tensor.reduction(full_tensor=full[i], red_tensor=red[i]) 
180   
181  # Set the reference domain. 
182  frame_order.ref_domain('N') 
183   
184  # Link the domains to the PDB files. 
185  frame_order.domain_to_pdb(domain='N', pdb='1J7O_1st_NH.pdb') 
186  frame_order.domain_to_pdb(domain='C', pdb='1J7P_1st_NH_rot.pdb') 
187   
188  # Set up the mechanics of the displacement to the average domain position. 
189  frame_order.average_position(pivot='motional', translation=False) 
190   
191  # Set the initial pivot point. 
192  pivot = array([ 37.254, 0.5, 16.7465]) 
193  frame_order.pivot(pivot, fix=True) 
194   
195  # Set the paramagnetic centre position. 
196  paramag.centre(pos=[35.934, 12.194, -4.206]) 
197   
198  # Duplicate the PCS data subset data pipe to create a data pipe containing all the PCS data. 
199  DATA = "Data - " + PIPE_BUNDLE 
200  pipe.copy(pipe_from=SUBSET, pipe_to=DATA, bundle_to=PIPE_BUNDLE) 
201  pipe.switch(DATA) 
202   
203  # Load the complete PCS data into the already filled data pipe. 
204  for i in range(len(ln)): 
205      pcs.read(align_id=ln[i], file=pcs_files[i], mol_name_col=1, res_num_col=2, spin_name_col=5, data_col=6, error_col=7) 
206   
207   
208   
209  # Execution. 
210  ############ 
211   
212  # Do not change! 
213  Frame_order_analysis(data_pipe_full=DATA, data_pipe_subset=SUBSET, pipe_bundle=PIPE_BUNDLE, grid_inc=GRID_INC, grid_inc_rigid=GRID_INC_RIGID, min_algor=MIN_ALGOR, num_int_pts_grid=NUM_INT_PTS_GRID, num_int_pts_subset=NUM_INT_PTS_SUBSET, func_tol_subset=FUNC_TOL_SUBSET, num_int_pts_full=NUM_INT_PTS_FULL, func_tol_full=FUNC_TOL_FULL, mc_sim_num=MC_NUM, mc_int_pts=MC_INT_PTS, mc_func_tol=MC_FUNC_TOL, models=MODELS)
