 26  If essential dependencies are missing, then an error message is printed and the program terminated. 
 27  """

 39  try:

 43      sys.exit()

179  # Relaxation curve fitting. 
180  try: 
181      from maths_fns.relax_fit import setup 
182      del setup 
183      C_module_exp_fn = True 
184  except ImportError, message: 
185      # The OS. 
186      system = platform.system() 
187   
188      # Does the compiled file exist. 
189      file = 'relax_fit.so' 
190      if system == 'Windows' or system == 'Microsoft': 
191          file = 'relax_fit.pyd' 
192      if not access('maths_fns' + sep + file, F_OK): 
193          C_module_exp_fn_mesg = "ImportError: relaxation curve fitting is unavailable, the corresponding C modules have not been compiled." 
194   
195      # Show the full error. 
196      else: 
197          C_module_exp_fn_mesg = "ImportError: " + message[0] + "\nRelaxation curve fitting is unavailable, try compiling the C modules." 
198

200      C_module_exp_fn = False
