+def gui_to_tuple(string):
+    """Convert the GUI obtained string to a tuple.
+
+    @param string:  The list in string form.
+    @type string:   str or unicode
+    @return:        The list.
+    @rtype:         list
+    """
+
+    # No value.
+    if string in ['', u'', None]:
+        return ()
+
+    # Convert.
+    val = eval(string)
+    if not isinstance(val, tuple):
+        val = tuple(val)
+
+    # Return the list.
+    return val
