  24  # Constant formulae 
  25  # ~~~~~~~~~~~~~~~~~ 
  26  # 
  27  #    Dipolar constants 
  28  #    ~~~~~~~~~~~~~~~~~ 
  29  #                           1   / mu0  \ 2  (gH.gN.h_bar)**2 
  30  #        dip_const_func  =  - . | ---- |  . ---------------- 
  31  #                           4   \ 4.pi /         <r**6> 
  32  # 
  33  # 
  34  #                             3   / mu0  \ 2  (gH.gN.h_bar)**2 
  35  #        dip_const_grad  =  - - . | ---- |  . ---------------- 
  36  #                             2   \ 4.pi /         <r**7> 
  37  # 
  38  # 
  39  #                           21   / mu0  \ 2  (gH.gN.h_bar)**2 
  40  #        dip_const_hess  =  -- . | ---- |  . ---------------- 
  41  #                           2    \ 4.pi /         <r**8> 
  42  # 
  43  # 
  44  #    CSA constants 
  45  #    ~~~~~~~~~~~~~ 
  46  #                           (wN.csa)**2 
  47  #        csa_const_func  =  ----------- 
  48  #                                3 
  49  # 
  50  #                           2.wN**2.csa 
  51  #        csa_const_grad  =  ----------- 
  52  #                                3 
  53  # 
  54  #                           2.wN**2 
  55  #        csa_const_hess  =  ------- 
  56  #                              3 
  57  # 
  58  #    Rex constants 
  59  #    ~~~~~~~~~~~~~ 
  60  #        rex_const_func  =  rhoex * (2.pi.wH)**2 
  61  # 
  62  #        rex_const_grad  =  (2.pi.wH)**2 
  63  # 
  64  #        rex_const_hess  =  0
