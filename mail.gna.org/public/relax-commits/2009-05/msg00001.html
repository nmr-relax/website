+        """Test the reading of an Xplor NOE restraints file fragment with a

+
+        # Execute the script.
+        self.relax.interpreter.run(script_file=sys.path[-1] +

+        restraints = [
+            ['@H28', '@H9',  3.0, 5.0],

+
+        # Alias the current data pipe.
+        cdp = pipes.get_pipe()
+

+            self.assertEqual(cdp.noe_restraints[i][1], restraints[i][1])
