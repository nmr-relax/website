+
+        # Open the dx.map user function window.
+        uf = uf_store['dx.map']

+        uf.create_wizard(parent=self.app.gui)
+

+        uf.page.uf_args['point'].selection_win_show()

+        uf.page.uf_args['point'].selection_win_data()
