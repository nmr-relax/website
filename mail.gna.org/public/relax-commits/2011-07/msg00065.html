+def is_mol_selected(selection=None):
+    """Query if the molecule is selected.
+
+    @keyword selection:     The molecule ID string.
+    @type selection:        str
+    """
+
+    # Find if any spins are selected.
+    select = False
+    for spin in spin_loop(selection):
+        if spin.select:
+            select = True
+            break
+
+    # Return the state.
+    return select
+
