+def get_values():
+    """Return a list of all the current frequencies.
+
+    @return:    The frequency list for the current data pipe.
+    @rtype:     list of float
+    """
+
+    # No frequency data.
+    if not hasattr(cdp, 'frq'):
+        return []
+
+    # The frequency values.
+    values = cdp.frq.values()
+
+    # Build a list of the unique frequencies.
+    frq = []
+    for value in values:
+        if value not in frq:
+            frq.append(value)
+
+    # Return the frqs.
+    return frq
