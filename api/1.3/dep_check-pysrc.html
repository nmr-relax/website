 23  # Module docstring. 
 24  """Module for checking relax dependencies. 
 25   
 26  If essential dependencies are missing, then an error message is printed and the program terminated. 
 27  """ 
 28   
 29  # Python modules. 
 30  import platform 
 31  from os import F_OK, access, sep 
 32  import sys 
 33   
 34   
 35  # Essential packages. 
 36  ##################### 
 37   
 38  # numpy. 
 39  try: 
 40      import numpy 
 41      del numpy 
 42  except ImportError: 
 43      sys.stderr.write("The dependency 'numpy' has not been installed.\n") 
 44      sys.exit() 
 45   
 46  # Command line option parser. 
 47  try: 
 48      import optparse 
 49      del optparse 
 50  except ImportError: 
 51      sys.stderr.write("The dependency 'Optik' has not been installed.\n") 
 52      sys.exit() 
 53   
 54  # Minfx python package check. 
 55  try: 
 56      import minfx 
 57      del minfx 
 58  except ImportError: 
 59      sys.stderr.write("The dependency 'minfx' has not been installed (see https://gna.org/projects/minfx/).\n") 
 60      sys.exit() 
 61   
 62   
 63  # Optional packages. 
 64  #################### 
 65   
 66  # Readline module. 
 67  try: 
 68      import readline 
 69      del readline 
 70      readline_module = True 
 71  except ImportError: 
 72      readline_module = False 
 73   
 74  # BZ2 compression module. 
 75  try: 
 76      import bz2 
 77      del bz2 
 78      bz2_module = True 
 79  except ImportError, message: 
 80      bz2_module = False 
 81      bz2_module_message = message.args[0] 
 82   
 83  # Gzip compression module. 
 84  try: 
 85      import gzip 
 86      del gzip 
 87      gzip_module = True 
 88  except ImportError, message: 
 89      gzip_module = False 
 90      gzip_module_message = message.args[0] 
 91   
 92  # Devnull. 
 93  try: 
 94      from os import devnull 
 95      del devnull 
 96      devnull_import = True 
 97  except ImportError, message: 
 98      devnull_import = False 
 99      devnull_import_message = message.args[0] 
100   
101  # Scientific Python import. 
102  try: 
103      from Scientific.IO import PDB 
104      del PDB 
105      scientific_pdb_module = True 
106  except ImportError: 
107      scientific_pdb_module = False 
108   
109  # VMD module imports. 
110  try: 
111      from Scientific.Visualization import VMD    # This requires Numeric to be installed (at least in Scientific 2.7.8). 
112      del VMD 
113      vmd_module = True 
114  except ImportError: 
115      vmd_module = False 
116   
117   
118  # Compiled C modules. 
119  ##################### 
120   
121  # Relaxation curve fitting. 
122  try: 
123      from maths_fns.relax_fit import setup 
124      del setup 
125      C_module_exp_fn = True 
126  except ImportError, message: 
127      # The OS. 
128      system = platform.system() 
129   
130      # Does the compiled file exist. 
131      file = 'relax_fit.so' 
132      if system == 'Windows' or system == 'Microsoft': 
133          file = 'relax_fit.pyd' 
134      if not access('maths_fns' + sep + file, F_OK): 
135          C_module_exp_fn_mesg = "ImportError: relaxation curve fitting is unavailable, the corresponding C modules have not been compiled." 
136   
137      # Show the full error. 
138      else: 
139          C_module_exp_fn_mesg = "ImportError: " + message[0] + "\nRelaxation curve fitting is unavailable, try compiling the C modules." 
140   
141      # Set the flag. 
142      C_module_exp_fn = False
