+        # Alias the current data pipe.
+        cdp = relax_data_store[relax_data_store.current_pipe]
+

+        if sim_index != None:
+            probs = cdp.probs_sim[sim_index]
+            alpha = cdp.alpha_sim[sim_index]
+            beta = cdp.beta_sim[sim_index]
+            gamma = cdp.gamma_sim[sim_index]
+

+        else:
+            probs = cdp.probs
