+        """The variable delay list loading GUI element.
+
+        @param event:   The wx event.
+        @type event:    wx event
+        """
+
+        # The file
+        filename = openfile(msg='Select VD file.', filetype='*.*', 
default='all files (*.*)|*')
+
+        # Abort if nothing selected
+        if not filename:
+            return
+
+        # Open the file
+        file = open(filename, 'r')
+
+        # Read entries
+        index = 0
+        for line in file:

+                t = float(line.replace('/n', ''))
+            except:
+                continue
+
