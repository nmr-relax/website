+def get_domain_ids():
+    """Return the list of all domain ID strings.
+
+    @return:        The list of all domain IDs.
+    @rtype:         list of str
+    """
+
+    # No pipe.

+
+    # No domain data.
+    if not hasattr(cdp, 'domain'):
+        return []
+
+    # The domain IDs, sorted.
+    ids = cdp.domain.keys()
+    ids.sort()
