685          def determine_rnd(self, model=None): 
686              """Function for returning the name of next round of optimisation.""" 
687       
688              # Get a list of all files in the directory model.  If no directory exists, set the round to 'init' or 0. 
689              try: 
690                  dir_list = listdir(resultsdir + model) 
691              except: 
692                  return 0 
693       
694              # Set the round to 'init' or 0 if there is no directory called 'init'. 
695              if 'init' not in dir_list: 
696                  return 0 
697       
698              # Create a list of all files which begin with 'round_'. 
699              rnd_dirs = [] 
700              for file in dir_list: 
701                  if search('^round_', file): 
702                      rnd_dirs.append(file) 
703       
704              # Create a sorted list of integer round numbers. 
705              numbers = [] 
706              for dir in rnd_dirs: 
707                  try: 
708                      numbers.append(int(dir[6:])) 
709                  except: 
710                      pass 
711              numbers.sort() 
712       
713              # No directories begining with 'round_' exist, set the round to 1. 
714              if not len(numbers): 
715                  return 1 
716       
717              # Determine the number for the next round (add 1 to the highest number). 
718              return numbers[-1] + 1
