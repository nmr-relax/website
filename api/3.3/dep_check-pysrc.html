 35 -def version_comparison(version1, version2): 

 36      """Compare software versions. 
 37   
 38      This will return: 
 39   
 40          - When version 1 is older, -1, 
 41          - When both versions are equal, 0, 
 42          - When version 1 is newer, 1. 
 43   
 44   
 45      @param version1:    The first version number. 
 46      @type version1:     str 
 47      @param version2:    The second version number. 
 48      @type version2:     str 
 49      @return:            The comparison result of the Python cmp() function applied to two lists of integers.  This will be one of [-1, 0, 1]. 
 50      @type return:       int 
 51      """ 
 52   
 53      # Strip out trailing zeros. 
 54      version1 = sub(r'(\.0+)*$', '', version1) 
 55      version2 = sub(r'(\.0+)*$', '', version2) 
 56   
 57      # Convert to a list of numbers. 
 58      version1 = [int(val) for val in version1.split('.')] 
 59      version2 = [int(val) for val in version2.split('.')] 
 60   
 61      # Return the comparison. 
 62      return (version1 > version2) - (version1 < version2)
