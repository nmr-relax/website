+    @type text:     str
+    @return:        The text with leading whitespace removed.

+    """
+
+    # Split by newline.
+    lines = split(text, '\n')
+
+    # Find the minimum whitespace.
+    min_white = 1000
+    for line in lines:
+        # Empty lines.
+        if strip(line) == '':
+            continue
+
+        # Count the whitespace for the current line.
