+

+        @type text:     str
+        @return:        The stripped text.
+        @rtype:         str
+        """
+
+        # First strip whitespace.
+        stripped_text = _strip_lead(text)
+

+        while 1:

+                break
+

+        while 1:
+            if stripped_text[-1] == "\n":
