 35      """Calculate and return the centre of mass of the structure. 
 36   
 37      @keyword return_mass:   A flag which if False will cause only the centre of mass to be returned, 
 38                              but if True will cause the centre of mass and the mass itself to be 
 39                              returned as a tuple. 
 40      @type return_mass:      bool 
 41      @return:                The centre of mass vector, and additionally the mass. 
 42      @rtype:                 list of 3 floats (or tuple of a list of 3 floats and one float) 
 43      """ 
 44   
 45      # Alias the current data pipe. 
 46      cdp = pipes.get_pipe() 
 47   
 48      # Test if a structure has been loaded. 
 49      if not hasattr(cdp, 'structure'): 
 50          raise RelaxNoPdbError 
 51   
 52      # Print out. 
 53      print "Calculating the centre of mass." 
 54   
 55      # Initialise the centre of mass. 
 56      R = zeros(3, float64) 
 57   
 58      # Initialise the total mass. 
 59      M = 0.0 
 60   
 61      # Loop over all atoms. 
 62      for mol_name, res_num, res_name, atom_num, atom_name, element, pos in cdp.structure.atom_loop(mol_name_flag=True, res_num_flag=True, res_name_flag=True, atom_num_flag=True, atom_name_flag=True, element_flag=True, pos_flag=True): 
 63          # Get the corresponding molecule container. 
 64          if mol_name == None: 
 65              mol_cont = cdp.mol[0] 
 66          else: 
 67              mol_cont = return_molecule('#' + mol_name) 
 68   
 69          # Get the corresponding residue container. 
 70          if res_name == None and res_num == None: 
 71              res_cont = mol_cont.res[0] 
 72          else: 
 73              res_cont = return_residue(':' + `res_num`) 
 74   
 75          # Get the corresponding spin container. 
 76          if atom_name == None and atom_num == None: 
 77              spin_cont = res_cont.spin[0] 
 78          else: 
 79              spin_cont = return_spin('@' + `atom_num`) 
 80   
 81          # Deselected spins. 
 82          if spin_cont and not spin_cont.select: 
 83              continue 
 84   
 85          # Atomic mass. 
 86          mass = return_atomic_mass(element) 
 87   
 88          # Total mass. 
 89          M = M + mass 
 90   
 91          # Sum of mass * position. 
 92          R = R + mass * pos 
 93   
 94      # Normalise. 
 95      R = R / M 
 96   
 97      # Final print out.

 99      print "    Centre of mass:  R = " + `R`
